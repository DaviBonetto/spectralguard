```python
import torch
import torch.nn.functional as F

class HiSPAv4:
    def __init__(self, target_model, ref_model, tokenizer, lambda_lex=0.5):
        self.target_model = target_model
        self.ref_model = ref_model # For perplexity/lexical constraints (e.g. GPT-2)
        self.tokenizer = tokenizer
        self.lambda_lex = lambda_lex

    def compute_lexical_loss(self, input_ids):
        # Calculate perplexity using the reference model
        with torch.no_grad():
            outputs = self.ref_model(input_ids, labels=input_ids)
            return outputs.loss # Cross Entropy Loss correlates with perplexity

    def compute_loss(self, input_ids, spectral_loss):
        lexical_loss = self.compute_lexical_loss(input_ids)
        total_loss = spectral_loss + self.lambda_lex * lexical_loss
        return total_loss, {'spectral_loss': spectral_loss.item(), 'lexical_loss': lexical_loss.item()}
```
